Ini script sederhana buat batch: Fetch top 10 coin, lalu detail + basic analysis (average sentiment). Pakai requests (atau SDK).

import requests
import json
from statistics import mean  # Untuk average

API_KEY = "YOUR_API_KEY"  # Ganti dengan key-mu
BASE_URL = "https://api.lunarcrush.com/v2"
HEADERS = {"Authorization": f"Bearer {API_KEY}"}

def fetch_top_coins(limit=10, sort="alt_rank"):
    url = f"{BASE_URL}/list/cryptocurrencies//{sort}/{limit}"
    resp = requests.get(url, headers=HEADERS)
    if resp.status_code == 200:
        return resp.json()["data"]
    else:
        print(f"Error: {resp.status_code}")
        return []

def fetch_coin_details(topic):
    url = f"{BASE_URL}/topic/{topic}"
    resp = requests.get(url, headers=HEADERS)
    if resp.status_code == 200:
        return resp.json()["data"]
    return None

# Batch Analysis: Top 10 coin
top_coins = fetch_top_coins(limit=10)
analysis = {}

for coin in top_coins:
    symbol = coin["symbol"].lower()  # e.g., "btc"
    details = fetch_coin_details(symbol)
    if details:
        analysis[symbol] = {
            "alt_rank": details.get("alt_rank"),
            "galaxy_score": details.get("galaxy_score"),
            "sentiment": details.get("sentiment"),
            "price": details.get("price"),
            "interactions_24h": details.get("interactions")
        }

# Contoh Output: Average Sentiment Semua Top Coin
sentiments = [data["sentiment"] for data in analysis.values() if data["sentiment"]]
avg_sentiment = mean(sentiments) if sentiments else 0
print(f"Batch Analysis - Top 10 Coins:")
print(json.dumps(analysis, indent=2))
print(f"Average Sentiment: {avg_sentiment:.2f} (Positif jika >0.5)")
