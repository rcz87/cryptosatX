import requests
import json

API_KEY = "YOUR_API_KEY"
BASE_URL = "https://api.lunarcrush.com/v2"

headers = {"Authorization": f"Bearer {API_KEY}"}

# Step 1: Get top 10 coins by AltRank
list_url = f"{BASE_URL}/list/cryptocurrencies//alt_rank/10"
list_resp = requests.get(list_url, headers=headers).json()
top_coins = [coin['symbol'] for coin in list_resp['data']]

# Step 2: Get detailed metrics for each
analysis = {}
for symbol in top_coins:
    topic_url = f"{BASE_URL}/topic/{symbol}"
    resp = requests.get(topic_url, headers=headers).json()
    data = resp['data']
    analysis[symbol] = {
        'galaxy_score': data.get('galaxy_score'),
        'sentiment': data.get('sentiment'),
        'percent_change_24h': data.get('percent_change_24h'),
        'market_cap': data.get('market_cap')
    }

print(json.dumps(analysis, indent=2))  # Output: Dict dengan metrics semua top coins
